public class Que1 {

	public static void main(String args[]) {
		int no1 = 101, no2 = 20;

		System.out.println("Enter 1 to add\n2 to substract\n3 To multiply\n4 To divide");

		int choice = 31;
		if (choice == 1) {
			int res = addition(no1, no2);
			System.out.println("Addition is" + res);
		} else if (choice == 2) {
			int res = substraction(no1, no2);
			System.out.println("Substraction is" + res);
		} else if (choice == 3) {
			int res = mulitplication(no1, no2);
			System.out.println("Multiplication is" + res);
		} else if (choice == 4) {
			int res = division(no1, no2);
			System.out.println("Division is" + res);
		} else
			System.out.println("invalid choice");
	}// main ends here

	static int addition(int no1, int no2) {

		return no1 + no2;

	}

	static int substraction(int no1, int no2) {

		return no1 - no2;

	}

	static int division(int no1, int no2) {

		return no1 / no2;

	}

	static int mulitplication(int no1, int no2) {

		return no1 * no2;

	}

}// class ends here
